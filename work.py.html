# определение функции декоратора
def choice_visitor(input_func):
    def output_func():  # определяем функцию, которая будет выполняться вместо оригинальной
        print("*"*48)  # перед выводом оригинальной функции выводим всякую звездочки
        input_func()  # вызов оригинальной функции
        print("*"*48)  # после вывода оригинальной функции выводим всякую звездочки

    return output_func  # возвращаем новую функцию

# создание списков

while True:

    standard_FIO = re.compile(r"^[А-Я][а-я]+(?:-[А-Я][а-я]+)*\s+[А-Я,Ё][а-я]"
                              r"+(?:-[А-Я][а-я]+)*\s+[А-Я,Ё][а-я]+(?:-[А-Я][а-я]+)*$")
    if standard_FIO.fullmatch(FIO_visitor):
        print("ФИО добавленно")
        data_visitor['ФИО'] = FIO_visitor
        break
    else:
        print("ФИО не верно!")

while True:
    try:
        age_visitor = int(input("Введите ваш возраст(от 0 до 120): "))
        if 0 < age_visitor < 120:

            data_visitor['возраст'] = age_visitor
            break
        else:
            print("Неправильный возраст!")

        print("Неправильное значение!")

# --3 Пользователь выбирает готовые варианты пицц, и напитки,
# либо выбирает кастомную пиццу с вариантами начинок.--
    print("\n---------ЗАКАЗ---------")

        print("1. Выбрать готовую пиццу\n"
              "2. Создать кастомную пиццу\n"
              "3. Выбрать напиток\n"
              "4. Завершить заказ")

        choice = input("Что выберите: ")

        if choice == "4":
            print("Заказ готовится")
            break

        elif choice == "1":
            print("\n------ВЫБОР ПИЦЦЫ------")
            for pizza, price in menu.items():
                print(f"{pizza} : {price['цена']}")
            user_pizza = input("Выберите пиццу: ")
            if user_pizza in menu.keys():
                print("Пицца добавлена")
                total_cost += menu[user_pizza]['цена']

            else:
                print("Пицца не добавлена!")
            order.append({
                "пицца": user_pizza,
                "цена": menu[user_pizza]['цена'],
                "тип": "пицца"
            })
            continue

        elif choice == "2":
            print("\n---------СОЗДАНИЕ КАСТОМНОЙ ПИЦЦЫ---------")
            base_price = 400
            custom_price = base_price
            selected_toppings = []
            while True:
                print("Доступные начинки:")
                for toppin, price in topping.items():
                    print(f"{toppin}: {price} руб.")
                print(f"\nБазовая цена пиццы: {base_price} руб.")
                print(f"\nТекущий состав: {", ".join(selected_toppings) if selected_toppings else "нет начинок"}\n"
                      f"Текущая цена: {custom_price} руб.")

                select = input("Введите:\n- название начинки чтобы её добавить\n- 'готово' для завершения: ").lower()

                if select == 'готово':
                    if not selected_toppings:
                        print("Вы не добавили ни одной начинки!")
                    break
                elif select in topping:
                    if select not in selected_toppings:
                        selected_toppings.append(select)
                        custom_price += topping[select]
                        print(f"Добавили: {select}")
                    else:
                        print(f"Начинка {select} уже добавлена")
                else:
                    print("Неизвестная команда или начинка")

            name_pizza = "Кастомная пицца"
            if selected_toppings:
                 name_pizza += f" ({", ".join(selected_toppings)})"#оператор SQL

            order.append({"название": name_pizza,
                          "\nцена": custom_price,
                          "\nтип": "пицца"})
            total_cost += custom_price
            print(f"\nКастомная пицца добавлена в заказ")
            continue

        elif choice == "3":
            print("\n---------Напитки----------")
            for drink, price in menu.items():
                print(f"{drink} : {price}")
            print("\n---------Напитки----------")
            drink_choice = input("Выберите напиток: \n")
            if drink_choice in menu:
                    total_cost += menu[drink_choice]
                    print("Ваш напиток добавлен")
                    order.append({"название": drink_choice,
                        "\nцена": menu[drink_choice],
                        "\nтип": "напитки"})

            else:
                print("Напиток не добавлен!")

        else:
            if not order:
                print("Ваш заказ пуст!")
            else:
                break
                print("Неверный выбор!")
